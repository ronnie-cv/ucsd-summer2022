# Compute summary metrics on data and match to clade info to obtain final dataframe
final_result = merge(chronum_input %>% 
                   dplyr::group_by(pangolin_lineage) %>% 
                   dplyr::summarize(min_date = min(date),
                                    max_date = max(date),
                                    nmins = sum(date == min(date)),
                                    nmembers = n()),
                 clade_info, 
                 by.x = "pangolin_lineage", by.y = "clade")

# Filter out incomplete dates
final_result = final_result %>% 
  dplyr::filter(nchar(min_date, type = "chars") == 10 &
                nchar(max_date, type = "chars") == 10) 

# Compute days from predicted emergence to first sampling
final_result$days = with(final_result, difftime(min_date, predicted_date, units = "days"))

# Compute days from first sampling to last sampling
final_result$clade_spread = with(final_result, difftime(max_date, min_date, units = "days"))

# Join with chronum_input to add country info
final_result = dplyr::inner_join(final_result, 
                                 chronum_input %>% dplyr::rename(min_date = date))

Results

Preliminary Country Analysis

To begin, it is interesting to view a distribution of the number of unique countries per clade. We can observe that generally, there is a larger quantity of clades with less than ~10 unique countries. This suggests that for many clades, sampling was confined to one country.

# Compute number of unique countries per clade
unique_countries_per_clade = chronum_input %>% dplyr::group_by(pangolin_lineage) %>% dplyr::summarise(unique_countries = length(unique(country))) 

# Plot
ggplot(unique_countries_per_clade,
       aes(x = unique_countries)) +
  geom_histogram(bins = 80) +
  labs(x = "Unique Countries",
       y = "Count",
       title = "Distribution of Unique Countries per Clade")

Detection Latency

We could additionally examine a potential relationship between the number of unique countries per clade, versus the days until the first sample, calculated as the delay in days between the first emergence of a clade, and the first observance (sample). Prior to generating this plot, the initial hypothesis was that clades with less unique countries would have a greater delay in reporting the first sample.

Rather, the plot indicates that there is a higher detection latency for countries with a smaller number of unique countries. We could infer that for clades with a greater number of unique countries, the processing time for the first sample is faster.

plot_df = inner_join(unique_countries_per_clade, final_result)
## Joining, by = "pangolin_lineage"
ggplot(plot_df %>% dplyr::filter(days >= 0),
       aes(x = unique_countries, y = as.numeric(days))) +
  geom_point() +
  labs(x = "Unique Countries Per Clade", y = "Days to First Sampling") +
  ggtitle("Days Until First Sample Based Upon No. of Unique Countries per Clade")

Clade Lifetime vs.Clade Size

Perhaps most interesting is examining the correlation between the lifetime of a clade (days elapsed from first to last sample report), and the size of the clade (represented by number of members in a clade). We could observe a significant positive correlation, as the increase in size tends to pair with an increase in days elapsed.

ggplot(final_result,
       aes(x = nmembers, y = as.numeric(clade_spread))) +
  geom_point() +
  scale_x_log10() +
  labs(x = "Clade Size", y = "Days From First to Last Reports") +
  ggtitle("Clade Lifetime vs. Clade Size") +
